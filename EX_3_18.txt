 #include <cstdlib>
#include <iostream>
using namespace std;
int main() {
 int i, j;

 i = 0;
 while (i <= 5) i++;
 cout << "A : i = " << i << endl; // i = 6

 i = j = 0;
 while (i <= 5) i += j++;
 cout << "B : i = " << i << " j = " << j << endl;
// 0: i = 0, j = 1
// 1: i = 1, j = 2
// 2: i = 3, j = 3
// 3: i = 6, j = 4

 i = j = 0;
 while (i <= 5) i += ++j;
 cout << "C : i = " << i << " j = " << j << endl;
// 0: j = 1, i = 1
// 1: j = 2, i = 3
// 2: j = 3, i = 6

 i = j = 0;
 while (j <= 5) i += j++;
 cout << "D : i = " << i << " j = " << j << endl;
 // 0: 
 
 i = j = 0;
 while (j <= 5) i += ++j;
 cout << "E : i = " << i << " j = " << j << endl;
 i = j = 0;
 while (i <= 5) i += 2; j++;
 cout << "F : i = " << i << " j = " << j << endl;

 return EXIT_SUCCESS;
}
